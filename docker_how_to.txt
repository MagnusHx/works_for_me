# Training image: build & run This project uses uv with a frozen lockfile (uv.lock) to make Python deps reproducible. ## Update lockfile (only when you change pyproject.toml)
bash
uv lock


# Build the training image (CPU / default platform)
docker build -f dockerfiles/train.dockerfile \
  --build-arg UV_GROUPS="--group train" \
  -t audio-emotion:train .

# Build as linux/amd64 (useful for CUDA wheels / deployment)

docker buildx build --platform linux/amd64 \
  -f dockerfiles/train.dockerfile \
  --build-arg UV_GROUPS="--group train" \
  -t audio-emotion:train \
  --load .

# Run Training
docker run --rm audio-emotion:train

# Run training with GPU
docker run --rm --gpus all audio-emotion:train
